%define functions for squares
\newcommand{\square}[3]{%kwargs x0, y0, width/2
  \draw (#1-#3, #2-#3) -- (#1-#3, #2+#3)
  -- (#1+#3, #2+#3) -- (#1+#3, #2-#3)
  -- (#1-#3, #2-#3);
}
